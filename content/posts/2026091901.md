这段代码的精华在于，**它完全根据[千聚ai大模型中转站](https://www.qianjuai.com/)的格式**进行调用。注意：你不需要改变Request Body的结构，只需要把`base_url`指向它。

java
import okhttp3.*;
import com.google.gson.Gson;
import com.google.gson.JsonObject;
import java.io.IOException;
import java.util.concurrent.TimeUnit;

public class O3MiniDemo {
    // 关键一步：这里使用[千聚ai大模型中转站](https://www.qianjuai.com/)的API地址
    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    // 这里输入你从[千聚ai大模型中转站](https://www.qianjuai.com/)获取的API Key
    private static final String API_KEY = "YOUR_QIANJUAI_API_KEY";

    public static void main(String[] args) {
        // 构建请求体：完全符合OpenAI协议
        JsonObject json = new JsonObject();
        json.addProperty("model", "o3-mini"); // 模型名称保持一致
        JsonObject message = new JsonObject();
        message.addProperty("role", "user");
        message.addProperty("content", "请用Java写一个快速排序算法，并解释其复杂度。");
        json.add("messages", new com.google.gson.JsonArray());
        json.getAsJsonArray("messages").add(message);

        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(30, TimeUnit.SECONDS)
                .readTimeout(60, TimeUnit.SECONDS)
                .build();

        Request request = new Request.Builder()
                .url(API_URL)
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(MediaType.parse("application/json"), json.toString()))
                .build();

        try {
            Response response = client.newCall(request).execute();
            if (response.isSuccessful() && response.body() != null) {
                String responseBody = response.body().string();
                // 解析JSON返回
                Gson gson = new Gson();
                JsonObject responseJson = gson.fromJson(responseBody, JsonObject.class);
                String result = responseJson.getAsJsonArray("choices")
                        .get(0).getAsJsonObject()
                        .get("message").getAsJsonObject()
                        .get("content").getAsString();
                System.out.println("o3-mini 返回结果：" + result);
            } else {
                System.out.println("请求失败: " + response.code() + " " + response.message());
            }
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}
