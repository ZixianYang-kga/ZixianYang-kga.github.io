import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.net.http.HttpRequest.BodyPublishers;
import java.net.http.HttpResponse.BodyHandlers;
import com.google.gson.Gson;
import com.google.gson.JsonObject;
import com.google.gson.JsonArray;

public class GPT4oAccessSample {

    // 千聚API服务地址
    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions";
    // 你的千聚API Key
    private static final String API_KEY = "YOUR_QIANJU_KEY";

    private static final HttpClient client = HttpClient.newHttpClient();
    private static final Gson gson = new Gson();

    public static void main(String[] args) throws Exception {
        String userMessage = "Hello, 用中文介绍一下你自己吧。";

        // 1. 非流式调用
        System.out.println("=== 非流式调用 ===");
        String nonStreamingResponse = callGPT4o(userMessage, false);
        System.out.println(nonStreamingResponse);

        // 2. 流式调用
        System.out.println("\n=== 流式调用（SSE源） ===");
        callGPT4oStream(userMessage);
    }

    // 非流式调用
    public static String callGPT4o(String userMessage, boolean stream) throws Exception {
        JsonObject requestBody = buildRequestBody(userMessage, stream);

        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_URL))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer " + API_KEY)
                .POST(BodyPublishers.ofString(gson.toJson(requestBody)))
                .build();

        HttpResponse<String> response = client.send(request, BodyHandlers.ofString());
        if (response.statusCode() != 200) {
            throw new RuntimeException("API调用失败: " + response.statusCode() + " " + response.body());
        }

        // 从JSON中提取回复
        JsonObject jsonResponse = gson.fromJson(response.body(), JsonObject.class);
        String content = jsonResponse
                .getAsJsonArray("choices")
                .get(0).getAsJsonObject()
                .getAsJsonObject("message")
                .get("content")
                .getAsString();

        return content;
    }

    // 流式调用（Server-Sent Events）
    public static void callGPT4oStream(String userMessage) throws Exception {
        JsonObject requestBody = buildRequestBody(userMessage, true);

        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create(API_URL))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer " + API_KEY)
                .POST(BodyPublishers.ofString(gson.toJson(requestBody)))
                .build();

        HttpResponse<String> response = client.send(request, BodyHandlers.ofString());

        if (response.statusCode() != 200) {
            throw new RuntimeException("流式API调用失败: " + response.statusCode() + " " + response.body());
        }

        // 流式响应通常包含多行"data:"前缀的事件
        String[] lines = response.body().split("\n");
        for (String line : lines) {
            if (line.startsWith("data:")) {
                String data = line.substring(5).trim();
                if (data.equals("[DONE]")) {
                    System.out.print("\n（流结束）");
                    break;
                }
                try {
                    JsonObject jsonObj = gson.fromJson(data, JsonObject.class);
                    // ChatGPT流式响应中，choices[0].delta.content
                    String delta = jsonObj
                            .getAsJsonArray("choices")
                            .get(0).getAsJsonObject()
                            .getAsJsonObject("delta")
                            .get("content")
                            .getAsString();
                    System.out.print(delta);
                } catch (Exception e) {
                    // 部分事件可能没有content字段,忽略
                }
            }
        }
    }

    private static JsonObject buildRequestBody(String userMessage, boolean stream) {
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "gpt-4o");
        requestBody.addProperty("stream", stream);

        JsonArray messages = new JsonArray();
        JsonObject systemMsg = new JsonObject();
        systemMsg.addProperty("role", "system");
        systemMsg.addProperty("content", "你是一位有用的助手。");
        JsonObject userMsg = new JsonObject();
        userMsg.addProperty("role", "user");
        userMsg.addProperty("content", userMessage);

        messages.add(systemMsg);
        messages.add(userMsg);
        requestBody.add("messages", messages);

        return requestBody;
    }
}
