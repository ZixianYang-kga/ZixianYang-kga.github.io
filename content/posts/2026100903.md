import com.alibaba.fastjson.JSONArray;
import com.alibaba.fastjson.JSONObject;

public class GrokRequestBuilder {

    public static String buildChatRequest(String userMessage) {
        JSONObject requestBody = new JSONObject();
        requestBody.put("model", "grok-1");
        requestBody.put("stream", true); // 开启流式响应，调用体验好
        requestBody.put("max_tokens", 2048);

        JSONArray messages = new JSONArray();
        JSONObject systemMessage = new JSONObject();
        systemMessage.put("role", "system");
        systemMessage.put("content", "你是一个有用的AI助手。");

        JSONObject userMsg = new JSONObject();
        userMsg.put("role", "user");
        userMsg.put("content", userMessage);

        messages.add(systemMessage);
        messages.add(userMsg);
        requestBody.put("messages", messages);

        return requestBody.toJSONString();
    }
}

### 第二步：核心调用方法（带避坑注释）

核心代码，注意我标注的避坑点：

java

import okhttp3.*;
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.concurrent.TimeUnit;

public class GrokApiClient {

    private static final String API_URL = "https://www.qianjuai.com/v1/chat/completions"; 
    private static final String API_KEY = "你的千聚API密钥"; // 替换成你的key

    public static void main(String[] args) {
        String message = "用中文回复我，简单介绍一下Java 21的新特性。";
        try {
            callGrokStream(message);
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public static void callGrokStream(String userMessage) throws IOException {
        // 避坑1: 一定要设置合理的超时时间, 国内网络环境建议30秒以上
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, TimeUnit.SECONDS)
                .readTimeout(60, TimeUnit.SECONDS)
                .writeTimeout(60, TimeUnit.SECONDS)
                .build();

        String jsonBody = GrokRequestBuilder.buildChatRequest(userMessage);

        RequestBody body = RequestBody.create(jsonBody, MediaType.parse("application/json; charset=utf-8"));

        Request request = new Request.Builder()
                .url(API_URL)
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .post(body)
                .build();

        // 避坑2: 不要用execute()方法, 要用enqueue(), 否则会阻塞UI线程
        // 这里为了演示清晰, 使用execute()并手动处理流
        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                System.err.println("请求失败, HTTP状态码: " + response.code());
                System.err.println("响应体: " + response.body().string());
                return;
            }

            // 避坑3: 这里必须用charStream, 不用byteStream
            // 因为SSE流返回的数据可能包含乱码, 用Reader处理安全
            BufferedReader reader = new BufferedReader(
                    new InputStreamReader(response.body().charStream(), "UTF-8"));

            String line;
            while ((line = reader.readLine()) != null) {
                // 避坑4: 要过滤掉空行和SSE的"data: [DONE]"标记
                if (line.isEmpty() || line.startsWith("data: [DONE]")) {
                    continue;
                }
                
                // 标准SSE格式: 每行以 "data: " 开头
                if (line.startsWith("data: ")) {
                    String jsonData = line.substring(6).trim();
                    try {
                        com.alibaba.fastjson.JSONObject jsonObj = 
                                com.alibaba.fastjson.JSONObject.parseObject(jsonData);
                        String content = jsonObj.getJSONArray("choices")
                                .getJSONObject(0)
                                .getJSONObject("delta")
                                .getString("content");
                        if (content != null) {
                            System.out.print(content); // 流式输出, 不换行
                        }
                    } catch (Exception e) {
                        // 避坑5: 解析错误直接跳过, 不要中断整个流
                        System.err.println("解析异常, 原始数据: " + jsonData);
                    }
                }
            }
        }
    }
}

## 接入避坑指南（最重要的部分）

我在调试这个接口时踩了 5 个坑，你写代码时务必注意：

### 1. 链接前缀不能写错

千万别写成 `https://www.qianjuai.com/v1/grok/chat` 之类的路径，服务是完全兼容 OpenAI 的接口标准，**路径必须是 `/v1/chat/completions`**。这是很多新手出问题最多的地方。

### 2. Header 必须用 Bearer 模式

java

.addHeader("Authorization", "Bearer sk-xxxxxxxxx")

千万不能用 `"AppKey: sk-xxx"` 或者直接 `"sk-xxx"` 这种写法，格式一错直接返回 401。

### 3. 流式输出的线程处理

如果你在 Spring Boot 的 Controller 里调这个接口，**记得用 `@Async` 注解或者用线程池**，因为流式输出会占用工作线程，处理不当容易把 Tomcat 线程池打满。

### 4. 不稳定的网络下要用重试机制

Grok4FastAPI 是国内直连，总体稳定，但偶发断流属于正常情况。建议你在调用时加一个重试逻辑，比如：

- 5 秒没有新数据，主动断开重连一次。
- 返回 429 或 503 状态码时，休眠 1 秒再重试。

### 5. 安全问题：不要硬编码密钥

实际生产环境，一定要把 API Key 放在环境变量里：
- Linux/Mac: `export GROK_API_KEY="sk-xxxxx"`
- 代码里: `System.getenv("GROK_API_KEY")`
- Spring Boot: 写在 `application.yml` 里用 `@Value` 注入

千万不要把 API Key 直接写在代码里提交到 Git 仓库！
