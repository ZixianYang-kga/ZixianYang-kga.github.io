import java.util.List;

public class ClaudeOpusCaller {

                ))
                .maxTokens(2048)
                .temperature(0.7)
                .build();

                .getChoices()
                .forEach(choice -> {
                    System.out.println("Claude Opus回复:");
                    System.out.println(choice.getMessage().getContent());
                });
    }
}
