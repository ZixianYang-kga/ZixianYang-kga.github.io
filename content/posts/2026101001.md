### 第二步：创建你的调用脚本

创建一个`qwen-demo.js`文件，复制以下代码：

javascript
const OpenAI = require('openai');

const client = new OpenAI({
  // 关键：把base_url改成千聚的地址，告别科学上网
  baseURL: "https://www.qianjuai.com/v1",
  apiKey: "sk-你的千聚APIKEY"  // 替换成你在千聚后台申请的Key
});

async function main() {
  console.log("🚀 正在调用通义千问 Qwen-Plus...");
  const response = await client.chat.completions.create({
    model: "qwen-plus",  // 这里想换什么模型就换什么标识符
    messages: [
      {
        role: "system",
        content: "你是一位精通通义千问API接入的Node.js专家。"
      },
      {
        role: "user",
        content: "你好，请用三句话介绍通义千问的主要能力。"
      }
    ],
    temperature: 0.7,
    max_tokens: 500
  });

  console.log("📝 回复内容：\n", response.choices[0].message.content);
  console.log("📊 Token用量：", {
    prompt: response.usage.prompt_tokens,
    completion: response.usage.completion_tokens,
    total: response.usage.total_tokens
  });
  console.log("✅ 😎 调用成功！");
}

main().catch(err => {
  console.error("❌ 调用出错：", err.message);
});
