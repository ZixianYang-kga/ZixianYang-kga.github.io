        // 我们为了更直观，用最简单的配置方式：
        // 在你的配置文件中设置环境变量，或者直接通过OkHttpClient拦截器。
        // 不过，更推荐的一个方案是直接使用千聚api提供的环境变量配置文件

    }

    public String chat(String userMessage) {

        String modelName = "gemini-2.5-flash";
