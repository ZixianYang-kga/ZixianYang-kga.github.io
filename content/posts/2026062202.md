**2. 编写Java代码**

import com.theokanning.openai.completion.CompletionRequest;
import com.theokanning.openai.completion.CompletionResult;
import com.theokanning.openai.service.OpenAiService;
import okhttp3.OkHttpClient;
import retrofit2.Retrofit;

import java.time.Duration;

public class Gpt5CodexExample {

    public static void main(String[] args) {
        // 关键步骤1：修改Base URL
        String baseUrl = "https://www.qianjuai.com/v1"; // 替换为千聚的API地址
        String apiKey = "sk-你的千聚API Key"; // 替换为你在千聚申请的Key

        // 关键步骤2：创建自定义的OkHttpClient
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(Duration.ofSeconds(30))
                .readTimeout(Duration.ofSeconds(60))
                .build();

        // 关键步骤3：构建服务
        OpenAiService service = new OpenAiService(apiKey, Duration.ofSeconds(60));

        // 关键步骤4：创建请求（假设我们调用gpt-4o作为兼容示例）
        CompletionRequest completionRequest = CompletionRequest.builder()
                .model("gpt-4o") // 或者 "gpt-5-codex-test"（如果模型已上线）
                .prompt("用Python写一个快速排序算法，并添加详细注释。")
                .maxTokens(500)
                .temperature(0.7)
                .build();

        // 发送请求
        CompletionResult result = service.createCompletion(completionRequest);
        System.out.println(result.getChoices().get(0).getText());
    }
}

- 把 `baseUrl` 从默认的 `https://api.openai.com/v1` 改成 `https://www.qianjuai.com/v1`。
