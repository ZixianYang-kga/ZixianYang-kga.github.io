### 第二步：Java代码调用Claude 3.5 Sonnet

java
import okhttp3.*;
import com.google.gson.JsonObject;
import com.google.gson.JsonParser;

import java.io.IOException;

public class ClaudeSonnetExample {

    private static final String API_KEY = "您的千聚API Key";
    // 关键：使用[千聚ai聚合站](https://www.qianjuai.com/)的统一API终点
    private static final String BASE_URL = "https://www.qianjuai.com/v1";

    public static void main(String[] args) throws IOException {
        OkHttpClient client = new OkHttpClient();

        // 构建请求体，指定model为claude-3-5-sonnet
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "claude-3-5-sonnet");
        JsonObject message = new JsonObject();
        message.addProperty("role", "user");
        message.addProperty("content", "用Java写一个单例模式的线程安全实现，并解释为什么这种实现比双重检查锁定更好。");
        requestBody.add("messages", new com.google.gson.JsonArray());
        requestBody.getAsJsonArray("messages").add(message);
        requestBody.addProperty("max_tokens", 1000);

        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .addHeader("Authorization", "Bearer " + API_KEY)
                .addHeader("Content-Type", "application/json")
                .post(RequestBody.create(
                        MediaType.parse("application/json"),
                        requestBody.toString()
                ))
                .build();

        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful() && response.body() != null) {
                String responseBody = response.body().string();
                JsonObject jsonResponse = JsonParser.parseString(responseBody).getAsJsonObject();
                String assistantReply = jsonResponse
                        .getAsJsonArray("choices")
                        .get(0)
                        .getAsJsonObject()
                        .get("message")
                        .getAsJsonObject()
                        .get("content")
                        .getAsString();
                System.out.println("Claude 3.5 Sonnet 回复：");
                System.out.println(assistantReply);
            } else {
                System.out.println("请求失败：" + response.code() + " " + response.message());
            }
        }
    }
}
