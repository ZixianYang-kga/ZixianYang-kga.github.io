## 完整Java代码——逐行注释教你调用Qwen-Turbo

下面就是完整的Java代码。不需要额外引入任何复杂的框架，只用Java标准库的 `HttpURLConnection` 就能完成API调用。每行代码后面我都写了中文注释，一边看一边写就能理解在做什么。

java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.io.OutputStream;
import java.net.HttpURLConnection;
import java.net.URL;
import java.nio.charset.StandardCharsets;

public class QwenTurboCaller {

    public static void main(String[] args) {
        // 第一步：定义请求的核心信息
        // 这是[千聚api聚合站](https://www.qianjuai.com/)的API接口地址，对应OpenAI的/v1/chat/completions
        String apiUrl = "https://www.qianjuai.com/v1/chat/completions";

        // 这是你在[千聚api聚合站](https://www.qianjuai.com/)控制台生成的API Key
        // 注意：实际使用时要把这个替换成你自己的Key
        String apiKey = "sk-你的钥匙在这里";

        // 我们这次要问模型的问题
        String userMessage = "用一句话解释什么是量子计算";

        try {
            // 第二步：创建URL对象，指向API接口
            URL url = new URL(apiUrl);
            
            // 第三步：打开连接，这是发送HTTP请求的核心对象
            HttpURLConnection connection = (HttpURLConnection) url.openConnection();
            
            // 第四步：设置请求方式——我们这次要发送数据给服务器，所以是POST
            connection.setRequestMethod("POST");
            
            // 第五步：设置请求头，告诉服务器我们发送的是JSON格式数据
            connection.setRequestProperty("Content-Type", "application/json");
            
            // 第六步：设置身份验证的请求头，格式是"Bearer 你的Key"
            // 这就是[千聚api聚合站](https://www.qianjuai.com/)识别你是谁，并且扣费的凭证
            connection.setRequestProperty("Authorization", "Bearer " + apiKey);
            
            // 第七步：允许向服务器发送数据（输出流），也允许从服务器接收数据（输入流）
            connection.setDoOutput(true);
            connection.setDoInput(true);

            // 第八步：构造要发送的JSON数据体
            // 这里指定了模型名称为qwen-turbo
            // messages数组里写了一个用户消息
            String jsonBody = "{"
                    + "\"model\": \"qwen-turbo\","
                    + "\"messages\": ["
                    + "  {\"role\": \"user\", \"content\": \"" + userMessage + "\"}"
                    + "]"
                    + "}";

            // 第九步：把JSON数据转成字节流，发送给服务器
            try (OutputStream os = connection.getOutputStream()) {
                byte[] input = jsonBody.getBytes(StandardCharsets.UTF_8);
                os.write(input, 0, input.length);
            }

            // 第十步：获取服务器的响应状态码
            // 200表示成功，其他数字表示有错误
            int responseCode = connection.getResponseCode();
            System.out.println("服务器响应状态码: " + responseCode);

            // 第十一步：根据状态码读取服务器返回的内容
            // 如果请求成功（200），从输入流读取；否则从错误流读取
            BufferedReader reader;
            if (responseCode == 200) {
                reader = new BufferedReader(
                        new InputStreamReader(connection.getInputStream(), StandardCharsets.UTF_8));
            } else {
                reader = new BufferedReader(
                        new InputStreamReader(connection.getErrorStream(), StandardCharsets.UTF_8));
            }

            // 第十二步：一行行读取响应内容，拼成一个完整的字符串
            StringBuilder response = new StringBuilder();
            String line;
            while ((line = reader.readLine()) != null) {
                response.append(line);
            }
            reader.close();

            // 第十三步：打印出服务器返回的完整JSON结果
            // 如果成功，里面会包含模型生成的回答
            System.out.println("模型返回结果: " + response.toString());

            // 第十四步：关闭连接，释放资源
            connection.disconnect();

        } catch (Exception e) {
            // 第十五步：如果过程中发生任何异常（比如网络不通、Key错误）
            // 这里会打印出错误信息，方便你排查
            System.out.println("调用过程中发生错误: " + e.getMessage());
            e.printStackTrace();
        }
    }
}

| 第7-9步 | 构造请求体，指定模型和消息内容 |
| 第10-13步 | 发起请求，读取响应 |
| 第14-15步 | 清理连接，异常处理 |
