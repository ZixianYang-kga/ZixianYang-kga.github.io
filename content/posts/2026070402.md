import java.time.Duration;
import java.util.Arrays;

public class Qwen3Demo {

    public static void main(String[] args) {
        // 替换为千聚分配的API Key
        String apiKey = "你的千聚API密钥";
        
        // 关键步骤：将base_url替换为千聚API地址
        String baseUrl = "https://www.qianjuai.com/v1";
        
        OpenAiService service = new OpenAiService(apiKey, Duration.ofSeconds(30));
        service.setBaseUrl(baseUrl);
        
        // 构造Qwen3请求，模型名字直接写 qwen3
        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("qwen3")
                .messages(Arrays.asList(
                        new ChatMessage("system", "你是一位Java开发专家"),
                        new ChatMessage("user", "请用Java示例代码解释调用Qwen3模型的步骤")
                ))
                .maxTokens(1024)
                .build();
                
        service.createChatCompletion(request).getChoices().forEach(choice -> {
            System.out.println(choice.getMessage().getContent());
        });
    }
}
