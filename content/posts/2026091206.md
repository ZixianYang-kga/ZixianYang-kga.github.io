  try {
    const response = await openai.chat.completions.create({
      model: 'deepseek-v3.2',  // 模型名称参照千聚控制台确认

        { role: 'user', content: message }
      ],
      max_tokens: 1024,
      temperature: 0.7,

    return null;
  }
}
